def least_squares_weight(x, y):
    
    xTx = x.T.dot(x)
    
    xTx_inv = np.linalg.inv(xTx)
    w = xTx_inv.dot(x.T.dot(y))
    y_hat = x.dot(w)
    error = np.mean((y - y_hat) ** 2)
    return w
